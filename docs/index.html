En pré-commit

  • .git/hooks/pre-commit
#!/bin/sh

eslint src/*.js test/js/*.js --ext json --ext js &&
rm -Rf docs/* && jsdoc src/ -c jdoc.json &&
rm -Rf build/* && ./build.sh

if [ $? -ne 0 ]; then
  echo "COMMIT ABORT!"
  exit 1
fi

git add -A .
exit 0
